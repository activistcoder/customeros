query GetContactById($contactId: ID!){
  contact(id: $contactId) {
    id
    conversations(sort:[{by:"STARTED_AT"}]) {
        totalPages
        totalElements
        content {
            id
            startedAt
            contactId
            userId
            contact {
                id
            }
            user {
                id
            }
        }
    }
  }
}
