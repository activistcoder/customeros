mutation UpdateCustomFieldInFieldSet($contactId: ID!, $fieldSetId: ID!, $fieldId: ID!){
     updateCustomFieldInFieldSet(
        contactId: $contactId,
        fieldSetId: $fieldSetId,
        input: {id: $fieldId
                datatype: TEXT
                name: "new name",
                value: "new value"}
     ) {
        id
        name
        value
        datatype
      }
}
